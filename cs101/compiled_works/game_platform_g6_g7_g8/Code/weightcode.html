function lbTokg() {
var lb = document.getElementById('Wlb').value
var result = lb * 0.453592
document.getElementById("Wresult").innerHTML = 'Result is ' + result.toFixed(1) + 'kg';
}
function kgTolb() {

document.getElementById("Wresults").innerHTML = 'Result is ' + result.toFixed(1) + 'lb';
}
