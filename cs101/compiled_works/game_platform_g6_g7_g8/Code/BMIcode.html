var height = document.getElementById('height').value
var weight = document.getElementById('weight').value
height /= 100
var result = (weight/(height*height))
if (Number(result) && result !== Infinity) {
document.getElementById('Bresult').innerHTML = result.toFixed(2)
}
if (result !== Infinity){
if (result < 18.5) {
document.getElementById('Bresults').innerHTML = 'You are underweight'
}
if (result > 18.5 && result < 24.9) {
document.getElementById('Bresults').innerHTML = 'You are healthy'
}
if (result > 24.9 && result < 29.9) { document.getElementById('Bresults').innerHTML = 'You are overweight'
}
if (result > 30) {
document.getElementById('Bresults').innerHTML = 'You are obese'
}
}
}
